96/// Description of a transaction, the unit of activity on Sui.
97#[Object]
98impl Transaction {
99    /// The transaction's globally unique identifier, which can be passed to `Query.node` to refetch it.
100    pub(crate) async fn id(&self) -> Id {
101        Id::Transaction(self.digest)
102    }
103
104    /// A 32-byte hash that uniquely identifies the transaction contents, encoded in Base58.
105    async fn digest(&self) -> String {
106        Base58::encode(self.digest)
107    }
108
109    /// The results to the chain of executing this transaction.
110    async fn effects(&self) -> Option<TransactionEffects> {
111        Some(TransactionEffects::from(self.clone()))
112    }
113
114    /// The type of this transaction as well as the commands and/or parameters comprising the transaction of this kind.
115    async fn kind(&self, ctx: &Context<'_>) -> Option<Result<TransactionKind, RpcError>> {
116        async {
117            let contents = self.contents.fetch(ctx, self.digest).await?;
118            let Some(content) = &contents.contents else {
119                return Ok(None);
120            };
121
122            let transaction_data = content.data()?;
123            Ok(TransactionKind::from(
124                transaction_data.kind().clone(),
125                contents.scope.clone(),
126            ))
127        }
128        .await
129        .transpose()
130    }
131
132    #[graphql(flatten)]
133    async fn contents(&self, ctx: &Context<'_>) -> Result<TransactionContents, RpcError> {
134        self.contents.fetch(ctx, self.digest).await
135    }
136}

158    /// This field is set by senders of a transaction block. It is an epoch reference that sets a deadline after which validators will no longer consider the transaction valid. By default, there is no deadline for when a transaction must execute.
159    async fn expiration(&self) -> Option<Result<Epoch, RpcError>> {
160        async {
161            let Some(content) = &self.contents else {
162                return Ok(None);
163            };
164
165            let transaction_data = content.data()?;
166            match transaction_data.expiration() {
167                TransactionExpiration::None => Ok(None),
168                TransactionExpiration::Epoch(epoch_id) => {
169                    Ok(Some(Epoch::with_id(self.scope.clone(), *epoch_id)))
170                }
171                TransactionExpiration::ValidDuring { max_epoch, .. } => {
172                    if let Some(epoch_id) = max_epoch {
173                        Ok(Some(Epoch::with_id(self.scope.clone(), *epoch_id)))
174                    } else {
175                        Ok(None)
176                    }
177                }
178            }
179        }
180        .await
181        .transpose()
182    }
183
184    /// The gas input field provides information on what objects were used as gas as well as the owner of the gas object(s) and information on the gas price and budget.
185    async fn gas_input(&self) -> Option<Result<GasInput, RpcError>> {
186        async {
187            let Some(content) = &self.contents else {
188                return Ok(None);
189            };
190
191            let transaction_data = content.data()?;
192            Ok(Some(GasInput::from_gas_data(
193                self.scope.clone(),
194                transaction_data.gas_data().clone(),
195            )))
196        }
197        .await
198        .transpose()
199    }

201    /// The address corresponding to the public key that signed this transaction. System transactions do not have senders.
202    async fn sender(&self) -> Option<Result<Address, RpcError>> {
203        async {
204            let Some(content) = &self.contents else {
205                return Ok(None);
206            };
207
208            let sender = content.data()?.sender();
209            Ok((sender != NativeSuiAddress::ZERO)
210                .then(|| Address::with_address(self.scope.clone(), sender)))
211        }
212        .await
213        .transpose()
214    }

218        async {
219            let Some(content) = &self.contents else {

229    /// The transaction as a JSON blob, matching the gRPC proto format (excluding BCS).
230    async fn transaction_json(&self) -> Option<Result<Json, RpcError>> {
231        async {
232            let Some(content) = &self.contents else {
233                return Ok(None);
234            };
235
236            let mut proto_transaction = content.proto_transaction()?;
237            // Clear the bcs field as transactionJson is intended to provide a full structured output
238            proto_transaction.bcs = None;
239            let json_value = serde_json::to_value(&proto_transaction)
240                .context("Failed to serialize transaction to JSON")?;
241            Ok(Some(json_value.try_into()?))
242        }
243        .await
244        .transpose()
245    }

254        Ok(signatures

262    /// Construct a transaction that is represented by just its identifier (its transaction
263    /// digest). This does not check whether the transaction exists, so should not be used to
264    /// "fetch" a transaction based on a digest provided as user input.
265    pub(crate) fn with_digest(scope: Scope, digest: TransactionDigest) -> Self {
266        Self {
267            digest,
268            contents: TransactionContents::empty(scope.with_active_transaction_digest(digest)),
269        }
270    }
271
272    /// Construct a fully-inflated transaction with already-hydrated contents. The digest is
273    /// read from `contents`, which keeps it consistent with the contents anchored on the scope.
274    pub(crate) fn with_contents(
275        scope: Scope,
276        contents: Arc<NativeTransactionContents>,
277    ) -> Result<Self, RpcError> {
278        let digest = contents.digest()?;
279        Ok(Self {
280            digest,
281            contents: TransactionContents {
282                scope: scope.with_active_transaction_contents(digest, contents.clone()),
283                contents: Some(contents),
284            },
285        })
286    }

288    /// Paginate over pre-loaded transactions, applying in-memory filtering.
289    ///
290    /// Used when transaction data is already available (e.g. from streaming) and doesn't
291    /// require database queries. Cursors encode `tx_sequence_number` for consistency with
292    /// the query API, enabling clients to continue paginating via queries.
293    ///
294    // TODO(DVX-2068): Add cursor consistency test between subscriptions and query API.
295    pub(crate) fn paginate_preloaded_transactions(
296        scope: Scope,
297        source_cp_sequence_number: u64,
298        transactions: &[ProcessedTransaction],
299        page: &Page<CTransaction>,
300        filter: TransactionFilter,
301    ) -> Result<TransactionConnection, RpcError> {
302        let after = page.after().map(|c| c.tx_sequence_number());
303        let before = page.before().map(|c| c.tx_sequence_number());
304
305        let mut filtered: Vec<_> = transactions
306            .iter()
307            .filter(|tx| filter.matches(&tx.contents))
308            .filter(|tx| after.is_none_or(|a| tx.tx_sequence_number >= a))
309            .filter(|tx| before.is_none_or(|b| tx.tx_sequence_number <= b))
310            .collect();
311
312        // `paginate_results` expects the window of results surrounding the requested page, so a
313        // `last` page keeps the tail of the filtered set, not the head.
314        if page.is_from_front() {
315            filtered.truncate(page.limit_with_overhead());
316        } else {
317            filtered.drain(..filtered.len().saturating_sub(page.limit_with_overhead()));
318        }
319
320        page.paginate_results(
321            filtered,
322            |tx| TransactionToken::cursor(source_cp_sequence_number, tx.tx_sequence_number),
323            |tx| Transaction::with_contents(scope.clone(), tx.contents.clone()),
324        )
325        .map(Into::into)
326        .map_err(upcast)
327    }
328
329    /// Load the transaction from the store, and return it fully inflated (with contents already
330    /// fetched). Returns `None` if the transaction does not exist (either never existed or was
331    /// pruned from the store).
332    pub(crate) async fn fetch(
333        ctx: &Context<'_>,
334        scope: Scope,
335        digest: Digest,
336    ) -> Result<Option<Self>, RpcError> {
337        let fetched = TransactionContents::empty(scope.clone())
338            .fetch(ctx, digest.into())
339            .await?;
340
341        let Some(contents) = fetched.contents else {
342            return Ok(None);
343        };
344
345        Ok(Some(Self::with_contents(scope, contents)?))
346    }

348    /// Cursor based pagination through transactions with filters applied.
349    ///
350    /// Returns empty results when no checkpoint is set in scope (e.g. execution scope).
351    pub(crate) async fn paginate(
352        ctx: &Context<'_>,
353        scope: Scope,
354        page: Page<CTransaction>,
355        filter: TransactionFilter,
356    ) -> Result<TransactionConnection, RpcError> {
357        let watermarks: &Arc<Watermarks> = ctx.data()?;
358        let available_range_key = AvailableRangeKey {
359            type_: "Query".to_string(),
360            field: Some("transactions".to_string()),
361            filters: Some(filter.active_filters()),
362        };
363        let reader_lo = available_range_key.reader_lo(watermarks)?;
364
365        let Some(query) = filter.tx_bounds(ctx, &scope, reader_lo, &page).await? else {
366            return Ok(TransactionConnection::empty());
367        };
368
369        let TransactionFilter {
370            after_checkpoint: _,
371            at_checkpoint: _,
372            before_checkpoint: _,
373            function,
374            kind,
375            affected_address,
376            affected_object,
377            sent_address,
378        } = filter;
379
380        let tx_sequence_numbers = if let Some(function) = function {
381            tx_call(ctx, query, &page, function, sent_address).await?
382        } else if let Some(kind) = kind {
383            tx_kind(ctx, query, &page, kind, sent_address).await?
384        } else if let Some(affected_object) = affected_object {
385            tx_affected_object(ctx, query, &page, affected_object, sent_address).await?
386        } else if let Some(address) = affected_address {
387            tx_affected_address(ctx, query, &page, address, sent_address).await?
388        } else if let Some(address) = sent_address {
389            tx_affected_address(ctx, query, &page, address, sent_address).await?
390        } else {
391            tx_unfiltered(ctx, query, &page).await?
392        };
393
394        page.paginate_results(
395            tx_digests(ctx, &tx_sequence_numbers).await?,
396            |(s, _)| TransactionToken::cursor(0, *s),
397            |(_, d)| Ok(Self::with_digest(scope.clone(), d)),
398        )
399        .map(Into::into)
400    }
401}

403impl TransactionContents {
404    fn empty(scope: Scope) -> Self {
405        Self {
406            scope,
407            contents: None,
408        }
409    }
410
411    /// Attempt to fill the contents. If the contents are already filled, returns a clone,
412    /// otherwise attempts to fetch from the store. The resulting value may still have an empty
413    /// contents field, because it could not be found in the store.
414    pub(crate) async fn fetch(
415        &self,
416        ctx: &Context<'_>,
417        digest: TransactionDigest,
418    ) -> Result<Self, RpcError> {
419        if self.contents.is_some() {
420            return Ok(self.clone());
421        }
422
423        // Reuse contents anchored on the scope by a parent resolver (streaming and indexed
424        // alike both anchor with hydrated contents when they have them).
425        if let Some(contents) = self.scope.active_transaction_contents_for(digest) {
426            return Ok(Self {
427                scope: self.scope.clone(),
428                contents: Some(contents.clone()),
429            });
430        }
431
432        let Some(checkpoint_viewed_at) = self.scope.checkpoint_viewed_at() else {
433            return Ok(self.clone());
434        };
435
436        let kv_loader: &KvLoader = ctx.data()?;
437        let Some(transaction) = kv_loader
438            .load_one_transaction(digest)
439            .await
440            .context("Failed to fetch transaction contents")?
441        else {
442            return Ok(self.clone());
443        };
444
445        // Discard the loaded result if we are viewing it at a checkpoint before it existed.
446        let cp_num = transaction
447            .cp_sequence_number()
448            .context("Any transaction fetched from the DB should have a checkpoint set")?;
449        if cp_num > checkpoint_viewed_at {
450            return Ok(self.clone());
451        }
452
453        Ok(Self {
454            scope: self.scope.clone(),
455            contents: Some(Arc::new(transaction)),
456        })
457    }
458}

568pub(crate) async fn tx_digests(
569    ctx: &Context<'_>,
570    tx_sequence_numbers: &[u64],
571) -> Result<Vec<(u64, TransactionDigest)>, RpcError> {
572    let pg_loader: &Arc<DataLoader<PgReader>> = ctx.data()?;
573
574    try_join_all(tx_sequence_numbers.iter().map(|&tx| async move {
575        let stored = pg_loader
576            .load_one(TxDigestKey(tx))
577            .await
578            .context("Failed to load transaction digest")?
579            .context("Failed to find transaction digest")?;
580
581        let digest = TransactionDigest::try_from(stored.tx_digest)
582            .context("Failed to deserialize transaction digest")?;
583
584        Ok((tx, digest))
585    }))
586    .await
587}

589async fn tx_affected_address(
590    ctx: &Context<'_>,
591    mut query: Query<'_>,
592    page: &Page<CTransaction>,
593    affected_address: SuiAddress,
594    sent_address: Option<SuiAddress>,
595) -> Result<Vec<u64>, RpcError> {
596    query += query!(
597        r#"
598        SELECT
599            tx_sequence_number
600        FROM
601            tx_affected_addresses
602        WHERE
603            affected = {Bytea}
604        "#,
605        affected_address.into_vec(),
606    );
607
608    if let Some(address) = sent_address {
609        query += query!(" AND sender = {Bytea}", address.into_vec());
610    }
611
612    tx_sequence_numbers(ctx, query, page).await
613}

641async fn tx_call(
642    ctx: &Context<'_>,
643    mut query: Query<'_>,
644    page: &Page<CTransaction>,
645    function: FqNameFilter,
646    sent_address: Option<SuiAddress>,
647) -> Result<Vec<u64>, RpcError> {
648    query += query!(
649        r#"
650        SELECT
651            tx_sequence_number
652        FROM
653            tx_calls
654        WHERE
655            package = {Bytea}
656        "#,
657        function.package().into_vec(),
658    );
659
660    if let Some(module) = function.module() {
661        query += query!(" AND module = {Text}", module.to_string());
662    }
663
664    if let Some(name) = function.name() {
665        query += query!(" AND function = {Text}", name.to_string());
666    }
667
668    if let Some(address) = sent_address {
669        query += query!(" AND sender = {Bytea}", address.into_vec());
670    }
671
672    tx_sequence_numbers(ctx, query, page).await
673}

675async fn tx_kind(
676    ctx: &Context<'_>,
677    mut query: Query<'_>,
678    page: &Page<CTransaction>,
679    kind: TransactionKindInput,
680    sent_address: Option<SuiAddress>,
681) -> Result<Vec<u64>, RpcError> {
682    match (kind, sent_address) {
683        // We can simplify the query to just the `tx_affected_addresses` table if ProgrammableTX
684        // and sender are specified.
685        (TransactionKindInput::ProgrammableTx, Some(address)) => {
686            tx_affected_address(ctx, query, page, address, Some(address)).await
687        }
688
689        (TransactionKindInput::SystemTx, Some(_)) => Ok(vec![]),
690
691        // Otherwise, we can ignore the sender always, and just query the `tx_kinds` table.
692        (_, None) => {
693            query += query!(
694                r#"
695                SELECT
696                    tx_sequence_number
697                FROM
698                    tx_kinds
699                WHERE
700                    tx_kind = {BigInt}
701                "#,
702                kind as i64,
703            );
704
705            tx_sequence_numbers(ctx, query, page).await
706        }
707    }
708}

712    mut query: Query<'_>,
713    page: &Page<CTransaction>,
714) -> Result<Vec<u64>, RpcError> {
715    query_limits::rich::debit(ctx)?;
716    let pg_reader: &PgReader = ctx.data()?;
717
718    query += query!(
719        r#"
720            AND (SELECT tx_lo FROM tx_lo) <= tx_sequence_number
721            AND tx_sequence_number < (SELECT tx_hi FROM tx_hi)
722        ORDER BY
723            tx_sequence_number {} /* order_by_direction */
724        LIMIT
725            {BigInt} /* limit_with_overhead */
726        "#,
727        page.order_by_direction(),
728        page.limit_with_overhead() as i64,
729    );
730
731    #[derive(QueryableByName)]
732    struct TxSequenceNumber {
733        #[diesel(sql_type = BigInt)]
734        tx_sequence_number: i64,
735    }
736
737    let mut conn = pg_reader
738        .connect()
739        .await
740        .context("Failed to connect to database")?;
741
742    let wrapped_tx_sequence_numbers: Vec<TxSequenceNumber> = conn
743        .results(query)
744        .await
745        .context("Failed to execute query")?;
746
747    let tx_sequence_numbers = if page.is_from_front() {
748        wrapped_tx_sequence_numbers
749            .iter()
750            .map(|t| t.tx_sequence_number as u64)
751            .collect()
752    } else {
753        wrapped_tx_sequence_numbers
754            .iter()
755            .rev()
756            .map(|t| t.tx_sequence_number as u64)
757            .collect()
758    };
759
760    Ok(tx_sequence_numbers)
761}

763/// The tx_sequence_numbers with cursors applied inclusively.
764/// Results are limited to `page.limit() + 2` to allow has_previous_page and has_next_page calculations.
765async fn tx_unfiltered(
766    ctx: &Context<'_>,
767    mut query: Query<'_>,
768    page: &Page<CTransaction>,
769) -> Result<Vec<u64>, RpcError> {
770    query_limits::rich::debit(ctx)?;
771    let pg_reader: &PgReader = ctx.data()?;
772
773    query += query!(
774        r#"
775        SELECT
776            (SELECT tx_lo FROM tx_lo),
777            (SELECT tx_hi FROM tx_hi)
778        "#
779    );
780
781    #[derive(QueryableByName)]
782    struct TxBounds {
783        #[diesel(sql_type = BigInt)]
784        tx_hi: i64,
785
786        #[diesel(sql_type = BigInt)]
787        tx_lo: i64,
788    }
789
790    let mut conn = pg_reader
791        .connect()
792        .await
793        .context("Failed to connect to database")?;
794
795    let results: Vec<TxBounds> = conn
796        .results(query)
797        .await
798        .context("Failed to execute query")?;
799
800    let (mut tx_lo, mut tx_hi) = results
801        .first()
802        .context("No valid checkpoints found")
803        .map(|bounds| (bounds.tx_lo as u64, bounds.tx_hi as u64))?;
804
805    let limit = page.limit_with_overhead() as u64;
806    if page.is_from_front() {
807        tx_hi = tx_hi.min(tx_lo.saturating_add(limit));
808    } else {
809        tx_lo = tx_lo.max(tx_hi.saturating_sub(limit));
810    }
811
812    let tx_sequence_numbers = (tx_lo..tx_hi).collect();
813    Ok(tx_sequence_numbers)
814}

880    /// Decode an edge cursor back into its `CursorToken`.
881    fn edge_token(cursor: &str) -> CursorToken {
882        match CTransaction::decode_cursor(cursor).expect("decodable edge cursor") {
883            CTransaction::Primary(c) => CursorToken::from(&*c),
884            CTransaction::Secondary(_) => panic!("expected grpc cursor, got legacy"),
885        }
886    }

888    fn edge_positions(conn: &TransactionConnection) -> Vec<u64> {
889        conn.edges
890            .iter()
891            .map(|e| match edge_token(&e.cursor).position {
892                Position::Transactions { tx_seq, .. } => tx_seq,
893                position => panic!("expected transactions position, got {position:?}"),
894            })
895            .collect()
896    }
